# Script to output CSV magnetometer data in target format
# Usage: Load this script in SerialKiller2 and run it
# The script will output each line from mag_data.csv as target=HX,HY,HZ;
# 
# NOTE: This script now uses the pre-generated mag_data_sk_script.txt
# To regenerate from CSV: python csv_to_sk_script.py <csv_file> scripts/mag_data_sk_script.txt

ptime=250;
@plot key-value --keys "HX,HY,HZ" --points 1000 --title 'Magnetometer Output Helmholtz Mag'

@info=Starting CSV magnetometer data output
@info=Loading pre-generated magnetometer data script...
@script -o mag_data_sk_script.txt
